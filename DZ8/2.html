2. Не выполняя кода, ответить, что выведет браузер и почему:


if (!("a" in window)) {
var a = 1;
}
alert(a);
а объявлена как глобальная var, поэтому для window она существует и результатом проверки ("a" in window) будет true. а поскольку у нас проверка !(), то для алерта она останется необъявленной, т.к. внутрь if мы не попадаем.
Результат: "a" будет undefined


var b = function a(x) {
x && a(--x);
};
alert(a);
а нигде не определена и не объявлялась
Результат: ошибка


function a(x) {
return x * 2;
}
var a;
alert(a);
В процессе чтения кода браузер в первую очередь производит поиск объявлений функций. Когда обнаруживает, создается функция и переменной, у которой имя будет совпадать с именем функции, присваивается ссылка на функцию.
Поскольку в этом коде переменной а не присвоено никакого значения, то присвоение функции не переопределено, и в "а" будет ссылка на функцию.
Результат: вывод функции


function b(x, y, a) {
arguments[2] = 10;
alert(a);
}
b(1, 2, 3);
В функцию передается 3 аргумента, затем 3й из них переопределяется. И он же выводится в алерт
Результат: 3


function a() {
alert(this);
}
a.call(null);
т.к. строгого режима не объявлялось, значения null и undefined будут заменены глобальным объектом. Глобальным будет объект window
https://developer.mozilla.org/ru/docs/Web/JavaScript/Reference/Global_Objects/Function/call
Результат: выведен объект window
